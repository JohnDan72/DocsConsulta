    const CourseType = new GraphQLObjectType({
        name: 'Course',
        fields: () => ({
            id: {type: GraphQLID},
            name: {type: GraphQLString},
            status: { type: GraphQLBoolean},
            professor: { //like populate
                type: require('./professor').ProfessorType,
                resolve(parent , args){
                    return Professor.findById( parent.professorId );
                }
            }
        })
    });
